163 void GameCanvas::keyPressed(QKeyEvent* pKeyEvent) {
164  // Supprimer ce premier test si la répétition de touche doit être signalée.
165  if (pKeyEvent->isAutoRepeat())
166  pKeyEvent->ignore();
167  else {
168  m_pGameCore->keyPressed(pKeyEvent->key());
169 
170  if (pKeyEvent->modifiers()==(Qt::ShiftModifier|Qt::ControlModifier)) {
171  switch (pKeyEvent->key()) {
172  case Qt::Key_I:
173  if (m_pDetailedInfosItem)
174  m_pDetailedInfosItem->setVisible(!m_pDetailedInfosItem->isVisible());
175  break;
176  case Qt::Key_P:
177  m_tickTimer.setInterval(m_tickTimer.interval()+1);
178  qDebug() << "Tick interval set to " << m_tickTimer.interval();
179  break;
180  case Qt::Key_M:
181  m_tickTimer.setInterval(m_tickTimer.interval()-1);
182  qDebug() << "Tick interval set to " << m_tickTimer.interval();
183  break;
184  }
