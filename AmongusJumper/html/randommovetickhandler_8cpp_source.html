53 void RandomMoveTickHandler::tick(long long elapsedTimeInMilliseconds) {
54  // Calcul de la distance parcourue par le sprite, selon sa vitesse et le temps écoulé.
55  double distance = elapsedTimeInMilliseconds * m_spriteVelocity / 1000.;
56 
57  // Détermine, selon l'angle, le déplacement en x et en y
58  double moveX = qCos(m_moveAngle) * distance;
59  double moveY = qSin(m_moveAngle) * distance;
60 
61  QPointF spriteMovement(moveX, moveY);
62 
63  bool collision = false;
64 
65  // Détermine la prochaine position du sprite
66  QRectF nextSpriteRect = m_pParentSprite->globalBoundingBox().translated(spriteMovement);
67 
68  // Si les collisions ne sont pas ignorées, on vérifie si le déplacement provoquerait une
69  // collision afin de modifier la trajectoire le cas échéant.
70  if (!m_ignoreCollision) {
71 
72  // Récupère tous les sprites de la scène que toucherait ce sprite à sa prochaine position
73  auto collidingSprites = m_pParentSprite->parentScene()->collidingSprites(nextSpriteRect);
74 
75  // Supprimer le sprite lui-même, qui collisionne toujours avec sa boundingbox
76  collidingSprites.removeAll(m_pParentSprite);
77 
78  collision = !collidingSprites.isEmpty();
79  }
80 
81  m_moveDuration -= elapsedTimeInMilliseconds;
82 
83  if (m_moveDuration < 0 ||
84  !m_pParentSprite->parentScene()->isInsideScene(nextSpriteRect) ||
85  collision) {
86  initNextMove();
87 
88  } else {
89  m_pParentSprite->setPos(m_pParentSprite->pos() + spriteMovement);
90  }
91 
92  // Si la destruction sur collision est activée, on vérifie si le sprite
93  // est en collision. Si c'est le cas, on le détruit.
94  if (m_destroyOnCollision && !m_pParentSprite->parentScene()->collidingSprites(m_pParentSprite).isEmpty()) {
95  // Il ne faut en aucun cas effacer immédiatement le sprite avec delete, car l'effacement immédiat du
96  // sprite provoque la destruction immédiate de ce handler, alors qu'il est en cours d'exécution. Cela
97  // provoquerait alors un crash.
98  // deleteLater() permet d'efface le sprite plus tard, une fois que le code qui le concerne a été
99  // complètement exécuté.
100  m_pParentSprite->deleteLater();
101  }
102 }
