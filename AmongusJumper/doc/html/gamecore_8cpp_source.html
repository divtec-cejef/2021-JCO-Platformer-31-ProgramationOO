187 void GameCore::tick(long long elapsedTimeInMilliseconds) {
188  Q_ASSERT(m_pEarth != nullptr);
189 
190  double radian = EARTH_SPEED * elapsedTimeInMilliseconds / 1000.;
191 
192  double newangle = m_pEarth->data(EARTH_ANGLE_KEY).toDouble() + radian;
193  double radius = m_pEarth->data(EARTH_RADIUS_KEY).toDouble();
194  QPointF newPos(std::cos(newangle)*radius, std::sin(newangle)*radius);

446 
447  QPixmap SkyPixmap(static_cast<int>(ClipRect.width()), static_cast<int>(ClipRect.height()));
448  SkyPixmap.fill(Qt::cyan);
449  Sprite* pSky = new Sprite(SkyPixmap);

451 
452  QPointF BallStartPos(ClipRect.center().x() - pFootBall->width()/2, ClipRect.top());
453

457  m_footballClipSpriteList << pFootBall << pGrass << pSky;
458 
459  QPropertyAnimation* pBallFallDownAnimation = new QPropertyAnimation(pFootBall, "y"); // La propriété à animer est y (la position y)
460  pBallFallDownAnimation->setDuration(1500);
461  pBallFallDownAnimation->setStartValue(BallStartPos.y());
462  pBallFallDownAnimation->setEndValue(BallStartPos.y() + ClipRect.height() - pFootBall->height());
463  pBallFallDownAnimation->setEasingCurve(QEasingCurve::OutBounce);
464 
465  QParallelAnimationGroup* pAllVanishAnimation = new QParallelAnimationGroup;
466  pAllVanishAnimation->addAnimation(createVanishAnimation(pFootBall));
467  pAllVanishAnimation->addAnimation(createVanishAnimation(pGrass));
468  pAllVanishAnimation->addAnimation(createVanishAnimation(pSky));
469 
470  QSequentialAnimationGroup* pClipAnimation = new QSequentialAnimationGroup;
471  pClipAnimation->addAnimation(pBallFallDownAnimation);
472  pClipAnimation->addAnimation(pAllVanishAnimation);
473 
474  m_pScene->addSpriteToScene(pFootBall);
475  m_pScene->addSpriteToScene(pGrass);
476  m_pScene->addSpriteToScene(pSky);
477

479 
480  pClipAnimation->start(QAbstractAnimation::DeleteWhenStopped);
